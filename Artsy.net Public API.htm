Getting Started

Let's make a few API calls to obtain the information for Andy Warhol.

1. You Already Have an Artsy Account

You have already signed up to Artsy. If you wish, you may create a separate developer account at artsy.net/sign_up.

2. Signed In

You're already signed into this developer website.

4. Get a Token

Obtain an authentication token that can be used to make API requests by POSTing to /api/tokens/xapp_token. Using curl: 
curl -v -X POST "https://api.artsy.net/api/tokens/xapp_token?client_id=f1080cb3d90afcc0497b&client_secret=9af69fed03c717b5a46000419d5e81a2"
With plain Ruby: 
require 'net/http'
require 'json'

client_id = 'f1080cb3d90afcc0497b'
client_secret = '9af69fed03c717b5a46000419d5e81a2'
api_url = URI.parse('https://api.artsy.net/api/tokens/xapp_token')
response = Net::HTTP.post_form(api_url, client_id: client_id, client_secret: client_secret)
xapp_token = JSON.parse(response.body)['token']
Using Hyperclient with Ruby: 
require 'hyperclient'

client_id = 'f1080cb3d90afcc0497b'
client_secret = '9af69fed03c717b5a46000419d5e81a2'

api = Hyperclient.new('https://api.artsy.net/api') do |api|
  api.headers['Accept'] = 'application/vnd.artsy-v2+json'
  api.headers['Content-Type'] = 'application/json'
  api.connection(default: false) do |conn|
    conn.use FaradayMiddleware::FollowRedirects
    conn.use Faraday::Response::RaiseError
    conn.request :json
    conn.response :json, content_type: /\bjson$/
    conn.adapter :net_http
  end
end

xapp_token = api.tokens.xapp_token._post(client_id: client_id, client_secret: client_secret).token
With Node.js or in the Browser using Superagent: 
var request = require('superagent');

var clientID = 'f1080cb3d90afcc0497b',
    clientSecret = '9af69fed03c717b5a46000419d5e81a2',
    apiUrl = 'https://api.artsy.net/api/tokens/xapp_token'
    xappToken;

request
  .post(apiUrl)
  .send({ client_id: clientID, client_secret: clientSecret })
  .end(function(res) {
    xappToken = res.body.token; 
  });

5. Make Requests

curl -v -L https://api.artsy.net/api/artists/andy-warhol -H 'X-Xapp-Token:JvTPWe4WsQO-xqX6Bts49ncJ49rIpehiED---Ch6By5grcRNbdqx-MVpXLELxLtIZwuuY1moMYdC8yvbucBxnWbjPDU-RS096pSbXrdEI823SVA0U1t2XSo57o--lPetZvow2eQEPl8qOzo30gb6Xb7GuBaJKauMKhXY2QqtUj1nKG-sKaPdVmROvIvaukDiKQWUVGo-wpQq2SwULGUX6hIzQCgtO4DVPOVwm4vFAoM='

Using Traverson with Node.js or in the Browser. 
var traverson = require('traverson'),
    JsonHalAdapter = require('traverson-hal'),
    xappToken = 'JvTPWe4WsQO-xqX6Bts49ncJ49rIpehiED---Ch6By5grcRNbdqx-MVpXLELxLtIZwuuY1moMYdC8yvbucBxnWbjPDU-RS096pSbXrdEI823SVA0U1t2XSo57o--lPetZvow2eQEPl8qOzo30gb6Xb7GuBaJKauMKhXY2QqtUj1nKG-sKaPdVmROvIvaukDiKQWUVGo-wpQq2SwULGUX6hIzQCgtO4DVPOVwm4vFAoM=';

traverson.registerMediaType(JsonHalAdapter.mediaType, JsonHalAdapter);
api = traverson.from('https://api.artsy.net/api').jsonHal();

api.newRequest()
  .follow('artist')
  .withRequestOptions({
    headers: {
      'X-Xapp-Token': xappToken,
      'Accept': 'application/vnd.artsy-v2+json'
    }
  })
  .withTemplateParameters({ id: 'andy-warhol' })
  .getResource(function(error, andyWarhol) {
    console.log(andyWarhol.name + 'was born in ' + andyWarhol.birthday + ' in ' + andyWarhol.hometown);
  });
